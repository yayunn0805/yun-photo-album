    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { 
            getAuth, 
            signInWithPopup, 
            GoogleAuthProvider, 
            onAuthStateChanged, 
            signOut 
        } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { 
            getFirestore, 
            collection, 
            addDoc, 
            onSnapshot, 
            serverTimestamp 
        } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        const firebaseConfig = {
            apiKey: "AIzaSyA-qw36OlMwhtkALOJRTyO-k0MSSVivog0",
            authDomain: "yunn-a3426.firebaseapp.com",
            projectId: "yunn-a3426",
            storageBucket: "yunn-a3426.firebasestorage.app",
            messagingSenderId: "255290205366",
            appId: "1:255290205366:web:9316ab84a82e0f81c1b217"
        };

        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);
        const provider = new GoogleAuthProvider();
        const appId = "yunn-a3426";

        const loginSection = document.getElementById('login-section');
        const userSection = document.getElementById('user-section');
        const btnLogin = document.getElementById('btn-login');
        const btnLogout = document.getElementById('btn-logout');
        const btnUpload = document.getElementById('btn-upload');
        const fileInput = document.getElementById('file-input');
        const imgPreview = document.getElementById('img-preview');
        const previewContainer = document.getElementById('preview-container');
        const recordsList = document.getElementById('records-list');
        const loader = document.getElementById('loader');

        let currentBase64 = "";
        let currentUser = null;
        let unsubscribeRecords = null;

        btnLogin.addEventListener('click', async () => {
            try {
                toggleLoader(true);
                await signInWithPopup(auth, provider);
            } catch (error) {
                console.error("Login error", error);
            } finally {
                toggleLoader(false);
            }
        });

        btnLogout.addEventListener('click', () => signOut(auth));

        onAuthStateChanged(auth, (user) => {
            currentUser = user;
            if (user) {
                loginSection.classList.add('hidden');
                userSection.classList.remove('hidden');
                document.getElementById('user-name').innerText = user.displayName;
                document.getElementById('user-email').innerText = user.email;
                document.getElementById('user-photo').src = user.photoURL;
                fetchRecords(user.uid);
            } else {
                loginSection.classList.remove('hidden');
                userSection.classList.add('hidden');
                if (unsubscribeRecords) unsubscribeRecords();
                recordsList.innerHTML = '<p class="text-center text-gray-300 text-sm py-8 font-medium">還沒有任何照片喔</p>';
            }
        });

        fileInput.addEventListener('change', (e) => {
            const file = e.target.files[0];
            if (!file) return;

            if (file.size > 800000) {
                alert("圖片太大了！請選擇小於 800KB 的圖片。");
                fileInput.value = "";
                return;
            }

            const reader = new FileReader();
            reader.onload = (event) => {
                currentBase64 = event.target.result;
                imgPreview.src = currentBase64;
                previewContainer.classList.remove('hidden');
                btnUpload.disabled = false;
            };
            reader.readAsDataURL(file);
        });

        btnUpload.addEventListener('click', async () => {
            if (!currentUser || !currentBase64) return;

            try {
                toggleLoader(true);
                const userPhotosCol = collection(db, 'artifacts', appId, 'users', currentUser.uid, 'photos');
                
                await addDoc(userPhotosCol, {
                    imageBase64: currentBase64,
                    createdAt: serverTimestamp(),
                    fileName: fileInput.files[0].name
                });

                fileInput.value = "";
                previewContainer.classList.add('hidden');
                btnUpload.disabled = true;
                currentBase64 = "";
                
            } catch (error) {
                console.error("Upload error", error);
            } finally {
                toggleLoader(false);
            }
        });

        function fetchRecords(uid) {
            const userPhotosCol = collection(db, 'artifacts', appId, 'users', uid, 'photos');
            unsubscribeRecords = onSnapshot(userPhotosCol, (snapshot) => {
                const docs = [];
                snapshot.forEach(doc => docs.push({ id: doc.id, ...doc.data() }));
                docs.sort((a, b) => (b.createdAt?.seconds || 0) - (a.createdAt?.seconds || 0));

                if (docs.length === 0) {
                    recordsList.innerHTML = '<p class="text-center text-gray-300 text-sm py-8 font-medium">還沒有任何照片喔</p>';
                    return;
                }

                recordsList.innerHTML = docs.map(doc => `
                    <div class="group flex items-center p-3 bg-white rounded-2xl border border-light-pink shadow-sm hover:shadow-md transition-all">
                        <div class="relative">
                            <img src="${doc.imageBase64}" class="w-14 h-14 object-cover rounded-xl bg-gray-100 border border-light-pink">
                            <div class="absolute inset-0 rounded-xl shadow-[inset_0_0_10px_rgba(252,197,204,0.2)]"></div>
                        </div>
                        <div class="ml-4 flex-1 min-w-0">
                            <p class="text-sm font-bold text-gray-600 truncate">${doc.fileName || '生活片段'}</p>
                            <p class="text-[10px] text-primary-pink font-medium uppercase tracking-tighter">
                                ${doc.createdAt ? new Date(doc.createdAt.seconds * 1000).toLocaleDateString() : '保存中...'}
                            </p>
                        </div>
                        <div class="opacity-0 group-hover:opacity-100 transition-opacity">
                            <svg xmlns="http://www.w3.org/2000/svg" class="h-4 w-4 text-primary-pink" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                                <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M4.318 6.318a4.5 4.5 0 000 6.364L12 20.364l7.682-7.682a4.5 4.5 0 00-6.364-6.364L12 7.636l-1.318-1.318a4.5 4.5 0 00-6.364 0z" />
                            </svg>
                        </div>
                    </div>
                `).join('');
            });
        }

        function toggleLoader(show) {
            loader.classList.toggle('active', show);
        }
    </script>
